                                            else -0.6 if -0.8<x<=-0.6
                                            else -0.4 if -0.6<x<=-0.4
                                            else -0.2 if -0.4<x<=-0.2
                                            else 0 if -0.2<x<=0
                                            else 0.2 if 0<x<=0.2
                                            else 0.4 if 0.2<x<=0.4
                                            else 0.6 if 0.4<x<=0.6
                                            else 0.8 if 0.6<x<=0.8
                                            else 1)
